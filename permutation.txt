class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        res = []
        def dfs(nums, current):
            if not nums:
                res.append(current)
                return
            for i in range(len(nums)):
                dfs(nums[:i] + nums[i+1:], current + [nums[i]])
        dfs(nums, [])
        return res
